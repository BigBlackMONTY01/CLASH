FIX 2 — BUILD THE 1V1 BACKEND (it doesn’t exist)
There is no 1v1 route file. Build it from scratch.
Create artifacts/api-server/src/routes/1v1.ts:

import { Router } from "express";
import { db } from "@workspace/db";
import { nanoid } from "nanoid";

const router = Router();

// In-memory room store (good enough for now)
const rooms: Record<string, {
  code: string;
  topic: string;
  creatorId: string;
  joinerId?: string;
  status: "waiting" | "active" | "done";
  createdAt: number;
}> = {};

// POST /api/1v1/create
router.post("/1v1/create", (req, res) => {
  try {
    const { topic, playerId } = req.body;
    if (!topic || !playerId) {
      res.status(400).json({ error: "topic and playerId required" });
      return;
    }
    const code = nanoid(6).toUpperCase();
    rooms[code] = { code, topic, creatorId: String(playerId), status: "waiting", createdAt: Date.now() };
    res.json({ code, topic, status: "waiting" });
  } catch (err) {
    console.error("1v1/create error:", err);
    res.status(500).json({ error: "Failed to create room" });
  }
});

// POST /api/1v1/join
router.post("/1v1/join", (req, res) => {
  try {
    const { code, playerId } = req.body;
    if (!code || !playerId) {
      res.status(400).json({ error: "code and playerId required" });
      return;
    }
    const room = rooms[code.toUpperCase()];
    if (!room) {
      res.status(404).json({ error: "Room not found. Check the code and try again." });
      return;
    }
    if (room.status !== "waiting") {
      res.status(409).json({ error: "This room is already in progress." });
      return;
    }
    if (room.creatorId === String(playerId)) {
      res.status(409).json({ error: "You can't join your own room." });
      return;
    }
    room.joinerId = String(playerId);
    room.status = "active";
    res.json({ code: room.code, topic: room.topic, status: room.status });
  } catch (err) {
    console.error("1v1/join error:", err);
    res.status(500).json({ error: "Failed to join room" });
  }
});

// GET /api/1v1/room/:code
router.get("/1v1/room/:code", (req, res) => {
  try {
    const room = rooms[req.params.code.toUpperCase()];
    if (!room) {
      res.status(404).json({ error: "Room not found" });
      return;
    }
    res.json(room);
  } catch (err) {
    console.error("1v1/room error:", err);
    res.status(500).json({ error: "Failed to get room" });
  }
});

export default router;


Then open artifacts/api-server/src/routes/index.ts and add:

import oneVOneRouter from "./1v1";
router.use(oneVOneRouter);


Install nanoid if not present: pnpm add nanoid in artifacts/api-server/.
On the frontend, make sure CREATE ROOM calls POST /api/1v1/create with { topic, playerId } and JOIN ROOM calls POST /api/1v1/join with { code, playerId }.
