    method: "POST",

      "Content-Type": "application/json",
      "anthropic-dangerous-direct-browser-access": "true",

  });

  const rawText = await response.text();

  if (!response.ok) {
    throw new Error(`HTTP ${response.status}\n${rawText.slice(0, 500)}`);
  }

  let data;
  try {
    data = JSON.parse(rawText);
  } catch (e) {
    throw new Error(`Could not parse response body:\n${rawText.slice(0, 500)}`);
  }

  if (data.error) {
    throw new Error(`API returned error:\n${JSON.stringify(data.error, null, 2)}`);
  }

  const textContent = (data.content || [])
    .filter(b => b.type === "text")
    .map(b => b.text)
    .join("");

  if (!textContent) {
    throw new Error(`No text content in response.\nFull response: ${JSON.stringify(data, null, 2).slice(0, 500)}`);
  }

  const start = textContent.indexOf("{");
  const end = textContent.lastIndexOf("}");

  if (start === -1 || end === -1) {
    throw new Error(`No JSON object found.\nModel said:\n${textContent.slice(0, 500)}`);
  }

  const jsonStr = textContent.slice(start, end + 1);
  let parsed;
  try {
    parsed = JSON.parse(jsonStr);
  } catch (e) {
    throw new Error(`JSON parse failed: ${e.message}\nJSON was:\n${jsonStr.slice(0, 500)}`);
  }

  setSimulation(parsed);
  setStatus("done");
} catch (err) {
  setError(err.message);
  setStatus("error");
}
